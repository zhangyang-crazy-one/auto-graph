export function normalizeInsets(input?: Insets | number): Insets;

	box: Box;

	box: Box;
	center: Point;
	anchors: AnchorPoint[];
	obstacleBox: Box;

export function getEdgePort(geometry: ShapeGeometry, toward: Point, preferredAnchor?: AnchorName): Point;
